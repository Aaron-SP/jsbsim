100  class Filter {
101  double prev_in;
102  double prev_out;
103  double ca;
104  double cb;
105  public:
106  Filter(void) {}
107  Filter(double coeff, double dt) {
108  prev_in = prev_out = 0.0;
109  double denom = 2.0 + coeff*dt;
110  ca = coeff*dt/denom;
111  cb = (2.0 - coeff*dt)/denom;
112  }
113  double execute(double in) {
114  double out = (in + prev_in)*ca + prev_out*cb;
115  prev_in = in;
116  prev_out = out;
117  return out;
118  }
119  };
